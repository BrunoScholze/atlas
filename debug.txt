## Apontamento de ordem de produção (fluxo completo)
Descrição: Funcionalidade mais utilizada do app. O fluxo começa na seleção do formulário e
termina no envio do apontamento ao backend. Existem duas variantes: por cronômetro e por
formulário manual. Dependendo da configuração do formulário, paradas de produção também
podem ser registradas nesse fluxo.

### Fluxo completo:
1. Usuário abre a lista de formulários disponíveis e seleciona um (form-list)
2. É levado para a tela de seleção de OP, onde:
   - Informa (ou lê via QR Code) a Ordem de Produção
   - Se o formulário exige máquina: seleciona a máquina
   - Se o formulário permite paradas (canReportStop=true): botão "Parada" aparece no rodapé
   - Se o formulário bloqueia apontamento manual (blockManualAppointment=true): botão "Avançar" é ocultado
   - Se for SFC e tiver cronômetro habilitado: botão de timer aparece no centro da tela
3a. Clicou no timer → vai para datasul-report-timer (apontamento por cronômetro)
3b. Clicou em Avançar → vai para datasul-report-v2 (apontamento por formulário manual)
3c. Clicou em Parada → registra uma parada de produção

export class DatasulOverheadLaborOrderSelectComponent implements OnInit {
    public literals: any = {};
    public isHideLoading: boolean = true;
    public productionOrder: DatasulProductionOrder = new DatasulProductionOrder();
    public selectedProductionOrder: Array<number> = [];
    private preventBack: Subscription;

    constructor(
        private poI18nService: PoI18nService,
        private alertCtrl: AlertController,
        private router: Router,
        private platform: Platform,
        private notificationService: PoNotificationService,
        private productionOrderService: DatasulProductionQueryService
    ) {
        const requests = [this.poI18nService.getLiterals(), this.poI18nService.getLiterals({ context: "productionOrderQuery" })];

        forkJoin(requests).subscribe((literals) => {
            literals.forEach((item) => Object.assign(this.literals, item));
        });
    }

    ionViewWillEnter() {
        if (this.selectedProductionOrder[0]) {
            this.productionOrder.productionOrderNumber = this.selectedProductionOrder[0];
            if (this.productionOrder.productionOrderNumber) {
                this.getProductionOrder();
            }
        }
    }

    getProductionOrder(): void {
        this.isHideLoading = false;
        this.productionOrderService.getProductionOrder(this.productionOrder.productionOrderNumber, "?").subscribe({
            next: (order) => {
                this.isHideLoading = true;
                this.productionOrder = order;

                if (this.productionOrder.statusCode < 8)
                    this.goToOverheadLaborListPage();
                else this.notificationService.error(this.literals.orderIsDoneError);
            },
            error: (error) => {
                console.log("###Erro getProductionOrder:", error);
                this.isHideLoading = true;
                throw error;
            }
        });
    }

    goToOverheadLaborListPage() {
        this.router.navigate([`overhead-labor-report/datasul/list/${this.productionOrder.productionOrderNumber}`], {
            state: {
                fromMenu: true,
            },
        });
        this.selectedProductionOrder = [];
    }

    goToAdvancedSearchPage() {
        this.router.navigate(["production-query/datasul"], {
            state: {
                zoomMode: true,
                selectedProductionOrder: this.selectedProductionOrder,
            },
        });
    }

    async readBarCode() {
        this.preventBack = this.platform.backButton.subscribeWithPriority(9999, () => { });

        let permission = await BarcodeScanner.checkPermissions();
        if (!permission) {
            permission = await BarcodeScanner.requestPermissions();

            if (!permission) {
                throw "Sem permiss�o para utilizar a c�mera";
            }
        }


        try {
            const barcodeData = await BarcodeScanner.scan();

            if (barcodeData) {
                // Leitura realizada com sucesso
                this.productionOrder.productionOrderNumber = Number(barcodeData.barcodes[0].displayValue);
                this.goToOverheadLaborListPage();
            }
        } catch (error) {
            // Erro na Leitura
            const alert = await this.alertCtrl
                .create({
                    header: this.literals.readingProblem,
                    message: this.literals.readingProblemMessage,
                    buttons: [
                        {
                            text: this.literals.cancel,
                        },
                        {
                            text: this.literals.toType,
                        },
                    ],
                })
                .finally(() => {
                    window.setTimeout(() => {
                        this.preventBack.unsubscribe();
                    }, 1000);
                });
            alert.present();
        }

    }
}
